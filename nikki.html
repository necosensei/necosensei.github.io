#include <boost/multiprecision/gmp.hpp>
#include <iostream>
using namespace std ;
using namespace boost ;
using namespace multiprecision ;
constexpr long n = 1000000000 + 3 ;
using nfloat = number<gmp_float<n> > ;
int main(){
  nfloat A = 1 ;
  nfloat B = nfloat(1) / sqrt(nfloat(2)) ;
  nfloat T = nfloat(1) / nfloat(4) ;
  nfloat X = 1 ;
  nfloat Y ;
  nfloat pi ;
  nfloat epsilon = pow(nfloat(10),-nfloat(n)) ;
  int m ;
  for(m = 0; abs(A - B) > epsilon; m++){
    Y = A ;
    A = ldexp((A+B),-1) ;
    B = sqrt(Y*B) ;
    Y-= A ;
    T-= (Y*Y)*X ;
    X = ldexp(X,1) ;
  }
  T = ldexp(T,2) ;
  Y = A+B ;
  pi = Y*Y/T ;
  cerr << m << endl ;
  cout << setprecision(n) << pi << endl;
  return 0 ;
}
